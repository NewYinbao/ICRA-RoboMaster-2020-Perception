244  BufferManager(std::shared_ptr<nvinfer1::ICudaEngine> engine, const int batchSize = 0,
245  const nvinfer1::IExecutionContext* context = nullptr)
246  : mEngine(engine)
247  , mBatchSize(batchSize)
248  {
249  for (int i = 0; i < mEngine->getNbBindings(); i++)
250  {
251  auto dims = context ? context->getBindingDimensions(i) : mEngine->getBindingDimensions(i);
252  size_t vol = context || !mBatchSize ? 1 : static_cast<size_t>(mBatchSize);
253  nvinfer1::DataType type = mEngine->getBindingDataType(i);
254  int vecDim = mEngine->getBindingVectorizedDim(i);
255  if (-1 != vecDim) // i.e., 0 != lgScalarsPerVector
256  {
257  int scalarsPerVec = mEngine->getBindingComponentsPerElement(i);
258  dims.d[vecDim] = divUp(dims.d[vecDim], scalarsPerVec);
259  vol *= scalarsPerVec;
260  }
261  vol *= volume(dims);
262  std::unique_ptr<ManagedBuffer> manBuf{new ManagedBuffer()};
263  manBuf->deviceBuffer = DeviceBuffer(vol, type);
264  manBuf->hostBuffer = HostBuffer(vol, type);
265  mDeviceBindings.emplace_back(manBuf->deviceBuffer.data());
266  mManagedBuffers.emplace_back(std::move(manBuf));
267  }
268  }
