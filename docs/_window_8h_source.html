87 static float IOU(window a, window b)
88 {
89  if( a.x1 > b.x1 && a.y1 > b.y1)
90  {
91  window temp;
92  temp = a;
93  a = b;
94  b = temp;
95  }
96  if(a.x2 < b.x1 || a.y2 < b.y1)return 0;
97  else
98  {
99  float dx = std::min(a.x2,b.x2) - std::max(a.x1,b.x1);
100  float dy = std::min(a.y2,b.y2) - std::max(a.y1,b.y1);
101 
102  return dx*dy/(a.width*a.height+b.width*b.height-dx*dy);
103  }
